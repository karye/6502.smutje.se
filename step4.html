Steg 4 — Interaktiv stegning, LCD och I/O

Mål

Lägg till manuell stegning via tryckknappar och LCD-display. Du kan stega igenom programmet en klockcykel eller en instruktion i taget — helt utan datorskärm.

Nya komponenter

AntalKomponent
1LCD1602 med I²C-backpack
2Tryckknappar
1Kopplingstråd (SYNC → D7)

Kopplingsschema

Fullt interaktivt schema: 5 blad

Arduino-kod

#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#define PHI2 2
#define RESB 4
#define RWB  3
#define SYNC 7
#define BTN_CLK   5
#define BTN_INSTR 6

LiquidCrystal_I2C lcd(0x27, 16, 2);
uint8_t mem[65536];

void setup() {
  Serial.begin(115200);
  pinMode(PHI2, OUTPUT); pinMode(RESB, OUTPUT);
  pinMode(RWB, INPUT); pinMode(SYNC, INPUT);
  pinMode(BTN_CLK, INPUT_PULLUP); pinMode(BTN_INSTR, INPUT_PULLUP);
  DDRK = 0x00; DDRF = 0x00;

  mem[0xFFFC]=0x00; mem[0xFFFD]=0x80;
  mem[0x8000]=0xEA; mem[0x8001]=0xEA; mem[0x8002]=0xEA;

  digitalWrite(RESB, LOW);
  for(int i=0;i<5;i++) pulse();
  digitalWrite(RESB, HIGH);
  for(int i=0;i<10;i++) pulse();

  lcd.init(); lcd.backlight();
  lcd.setCursor(0,0); lcd.print("W65C02S Ready");
}

void pulse() {
  digitalWrite(PHI2, LOW); delayMicroseconds(500);
  uint16_t a=(PINK<<0)|(PINF<<8); bool rw=digitalRead(RWB);
  if(rw){DDRA=0xFF;PORTA=mem[a];}else{DDRA=0x00;}
  digitalWrite(PHI2, HIGH); delayMicroseconds(500);
  if(!rw) mem[a]=PINA;
  lcd.setCursor(0,0); lcd.print("A:$"); lcd.print(a,HEX);
  lcd.setCursor(0,1); lcd.print("D:$"); lcd.print(mem[a],HEX);
}

void loop() {
  if(!digitalRead(BTN_CLK)){delay(50);while(!digitalRead(BTN_CLK));delay(50);pulse();}
  if(!digitalRead(BTN_INSTR)){delay(50);while(!digitalRead(BTN_INSTR));delay(50);
    do{pulse();}while(!digitalRead(SYNC));}
}

Testning

  1. LCD:n visar "W65C02S Ready"
  2. Knapp 1: en klockcykel — LCD uppdateras
  3. Knapp 2: kör tills nästa instruktion (SYNC)
