Mål

Lägg till manuell stegning via tryckknappar och en LCD-display som visar aktuell adress och instruktion. Nu kan du stega igenom programmet en klockcykel eller en instruktion i taget och se exakt vad CPU:n gör — utan datorskärm.

Arduino-kod

// Steg 4 — Manuell stegning + LCD-display
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

#define PHI2  2
#define RESB  4
#define RWB   3
#define SYNC  7
#define BTN_CLK   5
#define BTN_INSTR 6

LiquidCrystal_I2C lcd(0x27, 16, 2);  // Byt till 0x3F vid behov
uint8_t memory[65536];
bool clk_high = false;

void setup() {
  Serial.begin(115200);

  pinMode(PHI2, OUTPUT);
  pinMode(RESB, OUTPUT);
  pinMode(RWB,  INPUT);
  pinMode(SYNC, INPUT);
  pinMode(BTN_CLK,   INPUT_PULLUP);
  pinMode(BTN_INSTR, INPUT_PULLUP);

  DDRK = 0x00; DDRF = 0x00;

  // Testprogram
  memory[0xFFFC] = 0x00; memory[0xFFFD] = 0x80;
  memory[0x8000] = 0xEA;  // NOP
  memory[0x8001] = 0xEA;  // NOP
  memory[0x8002] = 0xEA;  // NOP

  // Reset
  digitalWrite(RESB, LOW);
  for (int i = 0; i < 5; i++) clockCycle();
  digitalWrite(RESB, HIGH);
  for (int i = 0; i < 10; i++) clockCycle();

  lcd.init(); lcd.backlight();
  lcd.setCursor(0, 0); lcd.print("W65C02S Ready");
}

void clockCycle() {
  digitalWrite(PHI2, LOW);
  delayMicroseconds(500);

  uint16_t addr = (PINK << 0) | (PINF << 8);
  bool rwb = digitalRead(RWB);

  if (rwb == HIGH) {
    DDRA = 0xFF; PORTA = memory[addr];
  } else {
    DDRA = 0x00;
  }

  digitalWrite(PHI2, HIGH);
  delayMicroseconds(500);

  if (rwb == LOW) memory[addr] = PINA;

  // Uppdatera LCD
  lcd.setCursor(0, 0);
  lcd.print("A:$"); lcd.print(addr, HEX); lcd.print("    ");
  lcd.setCursor(0, 1);
  lcd.print("D:$"); lcd.print(memory[addr], HEX);
  lcd.print(rwb ? " R" : " W");
}

void loop() {
  // Knapp 1: Enskild klockcykel
  if (digitalRead(BTN_CLK) == LOW) {
    delay(50);  // debounce
    while (digitalRead(BTN_CLK) == LOW);
    delay(50);
    clockCycle();
  }

  // Knapp 2: Kör tills nästa instruktion (SYNC hög)
  if (digitalRead(BTN_INSTR) == LOW) {
    delay(50);
    while (digitalRead(BTN_INSTR) == LOW);
    delay(50);
    do { clockCycle(); } while (digitalRead(SYNC) == LOW);
  }
}

Testning

  1. Ladda upp koden — LCD:n ska visa "W65C02S Ready"
  2. Tryck på Knapp 1 (Klocksteg): En klockcykel exekveras, LCD:n uppdateras med aktuell adress och data
  3. Tryck på Knapp 2 (Instr.steg): Klockan rullar tills SYNC går hög (ny instruktion laddad), sedan stannar den
  4. Databuss-LED:arna visar instruktionen binärt
  5. Klock-LED:en blinkar till vid varje klockpuls

Förväntat resultat

✅ Fullt interaktiv testbänk. Du kan stega genom programmet manuellt med knapparna. LCD:n visar adress och instruktion i realtid. Klock-LED och databuss-LED ger omedelbar visuell feedback — helt utan att titta på en datorskärm.
